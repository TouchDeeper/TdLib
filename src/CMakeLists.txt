        PCL/enveloping.cpp

# OpenCV
find_package(OpenCV REQUIRED)

#PCL

include_directories(${PCL_INCLUDE_DIRS})

        FileOperation/keyboard.h
        PCL/filter.h
        PCL/enveloping.h
        PCL/tools.h
        PCL/common_typedef.h
        )

        PUBLIC_HEADER
        "${HEADER_FILES}"

                    ${DEPENDENCIES}
                    ${OpenCV_LIBRARIES}
                    ${PCL_LIBRARIES}
                    )

        # IMPORTANT: Add the foo library to the "export-set"
        EXPORT ${PROJECT_NAME}Targets
